       ```python
           """验证 query_chat 配置，无效时记录警告"""
           import logging
           logger = logging.getLogger(__name__)
           
           if not self.query_chat.enabled:
               logger.info("Query Chat 功能已禁用（enabled=false）")
               return
           
           # 检查必要配置
           missing = []
           if not self.query_chat.api_key:
               missing.append("api_key")
           if not self.query_chat.base_url:
               missing.append("base_url")
           
           if missing:
               logger.warning(
                   f"QUERY_CHAT 配置不完整（缺少: {', '.join(missing)}），"
                   "query 清理功能将降级运行"
               )
       ```
